앱을 개발할 때 리소스들은 MainBundle을 기준으로 가져온다.

프레임워크에서 리소스를 사용해야할 때가 있는데 그땐 번들을 지정해줘야한다.

미리 번들을 정의해주면 편리하다.
```Swift
extension Bundle {
    /// 프레임워크 번들
    public static var frameworkBundle:Bundle{
        get{
            return Bundle(identifier: "번들이름")!
        }
    }
}
```

자주 사용하는 메소드들을 번들과 결합하여 사용하면 된다.
```Swift
extension UIStoryboard {
    public static func initFromFramework(_ sbName: String) -> UIStoryboard{
        return UIStoryboard.init(name: sbName, bundle: Bundle.frameworkBundle)
    }

    public static func initFromFramework(_ sbName:String, _ vcName:String) -> UIViewController {
        return UIStoryboard.init(name: sbName, bundle: Bundle.frameworkBundle).instantiateViewController(withIdentifier: vcName)
    }
}

extension UIImage {
    public static func frameworkImage(named: String) -> UIImage? {
        return UIImage(named: named, in: Bundle.frameworkBundle, compatibleWith: nil)
    }
}

extension UIFont {
    public static func registerFont(fileName: String, bundleIdentifier: String) {
        if let frameworkBundle = Bundle(identifier: bundleIdentifier) {
            let pathForResourceString = frameworkBundle.path(forResource: fileName, ofType: nil)
            let fontData = NSData(contentsOfFile: pathForResourceString!)
            let dataProvider = CGDataProvider(data: fontData!)
            let fontRef = CGFont(dataProvider!)
            var errorRef: Unmanaged<CFError>? = nil

            if (CTFontManagerRegisterGraphicsFont(fontRef, &errorRef) == false) {
                print("Failed to register font - register graphics font failed - this font may have already been registered in the main bundle.")
            }
        }
        else {
            print("Failed to register font - bundle identifier invalid.")
        }
    }
}
```
